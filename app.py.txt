from flask import Flask, render_template

app = Flask(__name__)

@app.route('/')
def home():
    return render_template('index.html')

@app.route('/about')
def about():
    return render_template('about.html')

@app.route('/projects')
def projects():
    projects = [
        {"title": "Flask Blog", "description": "A simple blog built with Flask and SQLite (prototype).", "link": "https://github.com/example/flask-blog"},
        {"title": "Weather App", "description": "Fetches weather using a public API and displays forecasts.", "link": "https://github.com/example/weather-app"},
        {"title": "Portfolio Site", "description": "This portfolio website showcasing projects and skills.", "link": "https://github.com/example/portfolio-site"},
    ]
    return render_template('projects.html', projects=projects)

@app.route('/contact')
def contact():
    return render_template('contact.html')

if __name__ == "__main__":
    app.run(debug=True)
